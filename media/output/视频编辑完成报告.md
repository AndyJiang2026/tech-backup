## 🔧 处理命令

```bash
ffmpeg -i video_vertical.mp4 \
  -i ../audio/horse_hooves.mp3 \
  -i ../audio/happy_bgm.mp3 \
  -filter_complex "
    [0:v]fade=t=in:st=0:d=1,fade=t=out:st=4:d=1,unsharp=5:5:1.2:5:5:0.3,eq=contrast=1.3:saturation=1.2:brightness=0.05,format=yuv420p[video];
    [1:a]volume=0.7[hooves];
    [2:a]atrim=0:5,afade=t=in:st=0:d=0.5,afade=t=out:st=4.5:d=0.5,volume=0.25[bgm];
    [hooves][bgm]amix=inputs=2:duration=first:dropout_transition=0.5[audio]
  " \
  -map "[video]" -map "[audio]" \
  -c:v libx264 -preset medium -crf 20 \
  -c:a aac -b:a 128k \
  video_final.mp4
```
